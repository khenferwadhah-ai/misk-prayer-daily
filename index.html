function highlightNextPrayer() { const rows = document.querySelectorAll(".row"); const now = new Date(); const currentMinutes = now.getHours() * 60 + now.getMinutes(); // إزالة الكلاس من كل الصفوف أولًا rows.forEach(r => r.classList.remove("next-prayer")); // إيجاد أول صلاة بعد الإمساك for (let row of rows) { const name = row.children[0].innerText.trim(); if (name === "الإمساك") continue; const time = row.children[1].innerText.trim(); const [h, m] = time.split(":").map(Number); const total = h * 60 + m; if (total > currentMinutes) { row.classList.add("next-prayer"); // تمييز الصف break; // نوقف عند أول صلاة قادمة } } } // تحديث كل 30 ثانية أو كل دقيقة setInterval(highlightNextPrayer, 30000); highlightNextPrayer();
